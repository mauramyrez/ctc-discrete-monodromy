\subsection{Hadamard parametrix in four dimensions}

As an analytic outlook, consider the renormalized two-point function for a
non-minimally coupled scalar field with Hadamard expansion
\begin{equation}
\label{eq:hadamard}
G^{(1)}(x,x')
= \frac{1}{8\pi^{2}}
\Bigl[
\frac{u(x,x')}{\sigma_{\epsilon}(x,x')}
+ v(x,x')\ln\!\Bigl(\frac{\sigma_{\epsilon}(x,x')}{\mu^{2}}\Bigr)
+ w(x,x')
\Bigr],
\end{equation}
where $\sigma(x,x')$ is the Synge world function (half the squared geodesic
interval), $\sigma_{\epsilon}=\sigma-i\epsilon$ regulates the coincidence
limit, and $u,v,w$ are smooth biscalar coefficients determined by the field
equation~\cite{DeWitt1975,BirrellDavies1982}. Near coincidence,
$\sigma=\tfrac12\,g_{\mu\nu}\xi^{\mu}\xi^{\nu}+O(|\xi|^{3})$ in Riemann
normal coordinates $\xi^{\mu}$.

For the non-minimal scalar-tensor coupling
$\mathcal{L}_{\mathrm{ST}}=\tfrac12(\partial\phif)^{2}+\zeta\,\phif R$,
variation yields the modified Klein--Gordon equation
\begin{equation}
\label{eq:modified-kg}
\bigl(\Box - m^{2} - \zeta R\bigr)\,\phif = 0,
\end{equation}
equivalent to a minimally coupled field with curvature-induced effective mass
\begin{equation}
\label{eq:m-eff}
m_{\mathrm{eff}}^{2}(x)=m^{2}+\zeta R(x)
\end{equation}
(distinct from the azimuthal mode mass $m_{\ast}$ of Sec.~\ref{sec:novikov}).
The Hadamard coefficient $u(x,x')$ acquires a curvature-dependent correction
$u\to u[1+\tfrac16\zeta R\,\sigma+O(\sigma^{2})]$, and the coincidence
pole $1/\sigma_{\epsilon}$ is \emph{proposed} to be regulated by the intrinsic
length scale
\begin{equation}
\label{eq:l-star-full}
\ell_{\star}
=\Bigl(\frac{\hbar\,|\zeta\,\phif|}{|R|_{\Sigma_{\mathrm{CH}}}}\Bigr)^{\!1/2},
\end{equation}
obtained by balancing the $\sigma^{-1}$ term against the
curvature-induced mass in~\eqref{eq:m-eff}.
This cutoff is an EFT modeling assumption, not a quantity computed by the
present numerical engine.

\subsection{\texorpdfstring{Null-winding truncation and a proposed bound on $\langle T_{\mu\nu}\rangle_{\mathrm{ren}}$}{Null-winding truncation and a proposed bound on renormalized stress-energy}}

Near the chronology horizon, null geodesics wind $n$ times about the throat
with proper-length scales $\Delta\ell_{n}\sim 2\pi n r_{\mathrm{CH}}$.
Kim--Thorne~\cite{KimThorne1991} show that each winding contributes
$(\Delta\ell_{n})^{-4}$ to $\langle\Tmn\rangle_{\mathrm{ren}}$.
Summing over windings,
\begin{equation}
\label{eq:winding-sum}
\sum_{n=1}^{\infty}(\Delta\ell_{n})^{-4}
\;\sim\;
\sum_{n=1}^{\infty}(2\pi n r_{\mathrm{CH}})^{-4}.
\end{equation}
In the minimal-coupling limit the ultraviolet coincidence scale
$\Delta\ell\to 0$ as the horizon is approached, producing the
$(\Delta\ell)^{-4}$ blue-shift pole independently of the convergent
$p$-series tail of~\eqref{eq:winding-sum}.

With the scalar-tensor cutoff~\eqref{eq:l-star-full}, the Hadamard
coincidence tube is smeared at radius $\ell_{\star}$: modes with
proper separation below $\ell_{\star}$ are suppressed by the modified
coefficient $v(x,x')$ carrying $\ln(\sigma_{\epsilon}/\mu^{2})$.
The number of resolvable windings about a circle of radius $r_{\mathrm{CH}}$
is therefore bounded by
\begin{equation}
\label{eq:n-max}
n_{\max}\sim \frac{2\pi r_{\mathrm{CH}}}{\ell_{\star}}
= 2\pi r_{\mathrm{CH}}
\Bigl(\frac{|R|_{\Sigma_{\mathrm{CH}}}}{\hbar\,|\zeta\,\phif|}\Bigr)^{\!1/2},
\end{equation}
and the regulated partial sum satisfies
\begin{equation}
\label{eq:truncated-sum}
\sum_{n=1}^{n_{\max}}(\Delta\ell_{n})^{-4}
\le \frac{1}{(2\pi r_{\mathrm{CH}})^{4}}
\sum_{n=1}^{n_{\max}}\frac{1}{n^{4}}
<\frac{\pi^{4}}{90\,(2\pi r_{\mathrm{CH}})^{4}}<\infty.
\end{equation}

\begin{proposition}[Proposed renormalized stress-energy bound; EFT outlook]
\label{prop:bounded-Tren}
Let $\phif$ satisfy~\eqref{eq:modified-kg} on the background~\eqref{eq:metric}
with $m_{\mathrm{eff}}^{2}>0$ in a neighborhood of the chronology horizon,
and assume a Hadamard parametrix of the form~\eqref{eq:hadamard} with
renormalization scale $\mu$.
Then, \emph{within the cutoff-regulated scalar-tensor sector defined by}
$\ell_{\star}$, the renormalized stress-energy is proposed to satisfy
\begin{equation}
\label{eq:master-bound}
\bigl\|
\langle\Tmn\rangle_{\mathrm{ren}}
\bigr\|_{\infty,\Sigma_{\mathrm{CH}}}
\le
\frac{\hbar}{4\pi^{2}\,\ell_{\star}^{4}}
=
\frac{|R|_{\Sigma_{\mathrm{CH}}}^{2}}{4\pi^{2}\,\hbar\,(\zeta\,\phif)^{2}}
<\infty,
\end{equation}
where $\ell_{\star}$ is given by~\eqref{eq:l-star-full}.
The null-winding sum~\eqref{eq:winding-sum} is truncated at
$n_{\max}$~\eqref{eq:n-max}, yielding the finite estimate~\eqref{eq:truncated-sum}.
This statement is an analytic EFT proposal; it is not numerically verified
by the Rust monodromy or geodesic engines of Sec.~\ref{sec:novikov}.
\end{proposition}

\begin{proof}[Sketch]
Substituting~\eqref{eq:hadamard} into the point-splitting formula
$\langle\Tmn\rangle_{\mathrm{ren}}=\lim_{x'\to x}
\mathcal{T}_{\mu\nu}[G^{(1)}(x,x')]$~\cite{BirrellDavies1982},
the $\sigma^{-2}$ coincidence divergence is canceled by the
$i\epsilon$ prescription and the renormalization of $v$.
The remaining $\sigma^{-1}$ contribution is regulated by
$m_{\mathrm{eff}}^{2}$ from~\eqref{eq:m-eff}: for $m_{\mathrm{eff}}^{2}>0$
the massive parametrix supplies the factor
$\exp(-m_{\mathrm{eff}}\sqrt{2\sigma})$ as $\sigma\to 0$.
On the chronology horizon, the Kim--Thorne ultraviolet pole arises from
the vanishing coincidence scale $\Delta\ell\to 0$; the scalar-tensor
smearing replaces $\Delta\ell$ by $\ell_{\star}$, bounding the local
contribution by $\hbar/(4\pi^{2}\ell_{\star}^{4})$.
Independently, the winding sum is truncated at $n_{\max}$~\eqref{eq:n-max},
so that~\eqref{eq:truncated-sum} remains finite.
Combining the ultraviolet cutoff with the truncated winding sum
motivates~\eqref{eq:master-bound} as an EFT bound.
\end{proof}

For a static observer in the chronology-respecting region,
\begin{equation}
\label{eq:rho-ren-bound}
\bigl|\langle\rho\rangle_{\mathrm{ren}}\bigr|
\le \hbar/(4\pi^{2}\ell_{\star}^{4}) < \infty
\end{equation}
is the corresponding formal density estimate under the same cutoff assumption.
